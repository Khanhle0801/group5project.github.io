        async function callGeminiAPI(prompt, systemInstruction) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                systemInstruction: { parts: [{ text: systemInstruction }] }

            };


            for (let i = 0; i < 3; i++) { // Reduced retries to 3 for snappier UI

                try {

                    const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });

                    if (!response.ok) throw new Error(`HTTP Error: ${response.status}`);

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text || "Xin lỗi, không thể tạo phản hồi lúc này.";

                } catch (error) {

                    if (i === 2) return `Hệ thống AI đang bận, vui lòng thử lại sau.`;

                    await new Promise(res => setTimeout(res, 1000 * (i + 1)));

                }

            }

        }

            const userText = chatInput.value.trim();

            if (!userText) return;

            

            addMessageUI(userText, 'user');

            chatInput.value = '';

            btnSendChat.disabled = true;


            // Typing Indicator

            const typingId = 'typing-' + Date.now();

            const typingWrapper = document.createElement('div');

            typingWrapper.id = typingId;

            typingWrapper.className = 'flex gap-2';

            typingWrapper.innerHTML = `

                <div class="w-8 h-8 rounded-full bg-gradient-to-br from-blue-500 to-violet-500 flex items-center justify-center text-white text-xs flex-shrink-0 shadow-sm">AI</div>

                <div class="bg-white border border-slate-200 p-3 rounded-2xl rounded-tl-none shadow-sm flex items-center h-[46px]">

                    <div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div>

                </div>

            `;

            chatMessages.appendChild(typingWrapper);

            chatMessages.scrollTop = chatMessages.scrollHeight;


            const reportText = getReportContext();

            const prompt = `Câu hỏi: "${userText}"\nTrả lời ngắn gọn, lịch sự bằng Tiếng Việt. Chỉ dựa vào dữ liệu trong báo cáo sau:\n${reportText}`;

            

            const responseText = await callGeminiAPI(prompt, "Bạn là chuyên gia tư vấn chiến lược doanh nghiệp.");

            

            document.getElementById(typingId).remove();

            

            // Format response

            const cleanResponse = responseText.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>').replace(/\n/g, '<br/>');

            addMessageUI(cleanResponse, 'ai');

            btnSendChat.disabled = false;

        }


        btnSendChat.addEventListener('click', handleChat);

        chatInput.addEventListener('keypress', (e) => {

            if (e.key === 'Enter') handleChat();

        });
